    <script>
        // Set Date
        const options = { month: 'long', day: 'numeric', year: 'numeric' };
        document.getElementById('current-date').innerText = new Date().toLocaleDateString("en-US", options);

        // Audio Logic
        let sequenceStarted = false;
        // Using a standard, reliable short beep sound
        const beep = new Audio('https://www.soundjay.com/buttons/button-35.mp3');

        function startBeepSequence() {
            if (!sequenceStarted) {
                sequenceStarted = true;
                
                // Play first beep immediately on click
                playBeep();
                
                // Beep exactly every 5 seconds to match the runner's turn
                setInterval(playBeep, 5000);
            }
        }

        function playBeep() {
            beep.currentTime = 0; // Reset to start in case of overlap
            beep.play().catch(e => console.log("Audio play blocked. Click the page again."));
        }
    </script>
