    <script>
        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        function playBeep(freq = 440, duration = 0.5) {
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain); gain.connect(audioCtx.destination);
            osc.frequency.value = freq; osc.start();
            gain.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime + duration);
            osc.stop(audioCtx.currentTime + duration);
        }

        let level = 1, shuttle = 1, running = false;
        let lSecs = 60, sSecs = 9.0;
        let sTimer, lTimer;
        const L_CIRC = 879.6; const S_CIRC = 754;

        const dropdown = document.getElementById('manual-record');
        for(let l=1; l<=21; l++) {
            for(let s=1; s<=16; s++) {
                let opt = document.createElement('option');
                opt.value = `${l}:${s}`;
                opt.innerHTML = `${l}:${s}`;
                dropdown.appendChild(opt);
            }
        }

        let bestL = localStorage.getItem('beepLevel') || 0;
        let bestS = localStorage.getItem('beepShuttle') || 0;
        initRecordDisplay();

        function initRecordDisplay() {
            if (bestL > 0) {
                dropdown.value = `${bestL}:${bestS}`;
                document.getElementById('trophy-icon').style.filter = "none";
                document.getElementById('trophy-icon').style.opacity = "1";
            }
        }

        function setManualRecord() {
            const val = dropdown.value.split(':');
            bestL = parseInt(val[0]);
            bestS = parseInt(val[1]);
            localStorage.setItem('beepLevel', bestL);
            localStorage.setItem('beepShuttle', bestS);
            initRecordDisplay();
        }

        function resetHistory() {
            if (confirm("Reset Record?")) {
                localStorage.clear();
                bestL = 0; bestS = 0;
                dropdown.value = "0:0";
                document.getElementById('trophy-icon').style.filter = "grayscale(1)";
                document.getElementById('trophy-icon').style.opacity = "0.5";
            }
        }

        function speak(text, rate = 1.4) {
            const utterance = new SpeechSynthesisUtterance(text);
            utterance.rate = rate; window.speechSynthesis.speak(utterance);
        }

        function handleStart() {
            if (running) return;
            document.getElementById('startBtn').style.visibility = 'hidden';
            let prepTime = 5;
            const prepOverlay = document.getElementById('prep-overlay');
            prepOverlay.style.display = 'block';
            const prepInterval = setInterval(() => {
                if (prepTime > 0) {
                    prepOverlay.innerText = prepTime;
                    speak(prepTime.toString(), 1.6);
                    playBeep(300, 0.1); prepTime--;
                } else {
                    clearInterval(prepInterval);
                    prepOverlay.style.display = 'none';
                    beginTest();
                }
            }, 1000);
        }

        function beginTest() {
            running = true;
            lTimer = setInterval(() => {
                if (lSecs > 0) {
                    lSecs -= 0.1;
                    document.getElementById('level-ring').style.strokeDashoffset = L_CIRC - (lSecs / 60) * L_CIRC;
                } else {
                    lSecs = 60; level++;
                    document.getElementById('level-txt').innerText = level;
                    playBeep(880, 0.8);
                    sSecs = Math.max(3.8, sSecs - 0.4); 
                }
            }, 100);
            runShuttle();
        }

        function runShuttle() {
            let timeLeft = sSecs;
            const sClock = document.getElementById('s-clock');
            playBeep(440, 0.4); 
            speak(`${level} ${shuttle}`);
            sTimer = setInterval(() => {
                timeLeft -= 0.1;
                sClock.innerText = Math.max(0, timeLeft).toFixed(1);
                document.getElementById('shuttle-ring').style.strokeDashoffset = S_CIRC - (timeLeft / sSecs) * S_CIRC;
                if (timeLeft <= 0) {
                    clearInterval(sTimer); shuttle++;
                    document.getElementById('shuttle-txt').innerText = shuttle;
                    runShuttle();
                }
            }, 100);
        }

        function finishTest() {
            running = false;
            clearInterval(sTimer); clearInterval(lTimer);
            window.speechSynthesis.cancel();
            
            let isNewBest = false;
            if (level > bestL || (level == bestL && shuttle > bestS)) {
                localStorage.setItem('beepLevel', level);
                localStorage.setItem('beepShuttle', shuttle);
                isNewBest = true;
                bestL = level; bestS = shuttle;
                initRecordDisplay();
            }

            document.getElementById('final-display').innerText = `${level}:${shuttle}`;
            if (isNewBest) document.getElementById('new-record-banner').style.display = 'block';
            document.getElementById('overlay').style.display = 'flex';
        }
    </script>
