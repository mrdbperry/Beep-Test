    <script>
        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        let level = 1, shuttle = 1, currentCat = "", levelIndex = 0;
        let sSecs = 9.0, sTimer, globalTimer;
        let totalElapsedSeconds = 0;
        let totalShuttlesCompleted = 0;
        const music = document.getElementById('bg-music');

        const BEEP_DATA = [
            { level: 1, shuttles: 7, speed: 8.0 }, { level: 2, shuttles: 8, speed: 9.0 },
            { level: 3, shuttles: 8, speed: 9.5 }, { level: 4, shuttles: 9, speed: 10.0 },
            { level: 5, shuttles: 9, speed: 10.5 }, { level: 6, shuttles: 10, speed: 11.0 },
            { level: 7, shuttles: 10, speed: 11.5 }, { level: 8, shuttles: 11, speed: 12.0 },
            { level: 9, shuttles: 11, speed: 12.5 }, { level: 10, shuttles: 11, speed: 13.0 },
            { level: 11, shuttles: 12, speed: 13.5 }, { level: 12, shuttles: 12, speed: 14.0 },
            { level: 13, shuttles: 13, speed: 14.5 }, { level: 14, shuttles: 13, speed: 15.0 },
            { level: 15, shuttles: 13, speed: 15.5 }, { level: 16, shuttles: 14, speed: 16.0 },
            { level: 17, shuttles: 14, speed: 16.5 }, { level: 18, shuttles: 15, speed: 17.0 },
            { level: 19, shuttles: 15, speed: 17.5 }, { level: 20, shuttles: 16, speed: 18.0 },
            { level: 21, shuttles: 16, speed: 18.5 }
        ];

        function updateVolume() { music.volume = document.getElementById('volSlider').value; }

        function initRecords() {
            ['JNR_GIRLS', 'JNR_BOYS', 'SNR_GIRLS', 'SNR_BOYS', 'OPEN_AGES'].forEach(cat => {
                const stored = localStorage.getItem('beep_record_' + cat) || "1:1";
                const pbEl = document.getElementById('pb-' + cat);
                if(pbEl) pbEl.innerText = 'PB: ' + stored;
            });
            updateVolume();
        }

        const dropdown = document.getElementById('manual-record');
        for(let l=1; l<=21; l++) {
            const shuts = BEEP_DATA[l-1].shuttles;
            for(let s=1; s<=shuts; s++) {
                let opt = document.createElement('option'); 
                opt.value = `${l}:${s}`; opt.innerHTML = `${l}:${s}`; 
                dropdown.appendChild(opt);
            }
        }

        function selectCat(catId) {
            currentCat = catId;
            dropdown.value = localStorage.getItem('beep_record_' + currentCat) || "1:1";
            document.getElementById('active-cat-title').innerText = catId.replace('_', ' ');
            document.getElementById('category-view').classList.remove('active');
            document.getElementById('timer-view').classList.add('active');
        }

        function setManualRecord() {
            localStorage.setItem('beep_record_' + currentCat, dropdown.value);
            initRecords();
        }

        function formatTime(seconds) {
            const m = Math.floor(seconds / 60);
            const s = Math.floor(seconds % 60);
            return `${m.toString().padStart(2, '0')}:${s.toString().padStart(2, '0')}`;
        }

        function playBeep(freq = 440, duration = 0.5) {
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain); gain.connect(audioCtx.destination);
            osc.frequency.value = freq;
            gain.gain.setValueAtTime(1.5, audioCtx.currentTime); // High-gain boost
            gain.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime + duration);
            osc.start(); osc.stop(audioCtx.currentTime + duration);
        }

        function speak(text) {
            const utt = new SpeechSynthesisUtterance(text);
            utt.volume = 1.0; utt.rate = 1.7; utt.pitch = 1.2;
            window.speechSynthesis.speak(utt);
        }

        function handleStart() {
            document.getElementById('startBtn').style.visibility = 'hidden';
            music.play().catch(e => console.log("Music play blocked"));
            let prep = 5;
            const overlay = document.getElementById('prep-overlay');
            overlay.style.display = 'block';
            const int = setInterval(() => {
                if(prep > 0) { 
                    overlay.innerText = prep; playBeep(300, 0.1); speak(prep); prep--; 
                } else { 
                    clearInterval(int); overlay.style.display = 'none'; startTest(); 
                }
            }, 1000);
        }

        function startTest() {
            levelIndex = 0; level = BEEP_DATA[levelIndex].level; shuttle = 1;
            totalElapsedSeconds = 0;
            totalShuttlesCompleted = 0;
            globalTimer = setInterval(() => {
                totalElapsedSeconds++;
                document.getElementById('elapsed-clock').innerText = formatTime(totalElapsedSeconds);
            }, 1000);
            updateUI(); runShuttle();
        }

        function runShuttle() {
            const data = BEEP_DATA[levelIndex];
            sSecs = 20 / (data.speed / 3.6);
            let timeLeft = sSecs;
            playBeep(440, 0.4); speak(`${level} ${shuttle}`);
            
            sTimer = setInterval(() => {
                timeLeft -= 0.1;
                document.getElementById('s-clock').innerText = Math.max(0, timeLeft).toFixed(1);
                document.getElementById('shuttle-ring').style.strokeDashoffset = 754 - (timeLeft/sSecs)*754;
                if(timeLeft <= 0) {
                    clearInterval(sTimer);
                    totalShuttlesCompleted++;
                    if(shuttle < data.shuttles) {
                        shuttle++; runShuttle();
                    } else {
                        levelIndex++;
                        if(levelIndex < BEEP_DATA.length) {
                            level = BEEP_DATA[levelIndex].level; shuttle = 1;
                            playBeep(880, 0.8); runShuttle();
                        } else { finishTest(); }
                    }
                    updateUI();
                }
            }, 100);
        }

        function updateUI() {
            document.getElementById('level-txt').innerText = level;
            document.getElementById('shuttle-txt').innerText = shuttle;
            document.getElementById('level-ring').style.strokeDashoffset = 879.6 - (level/21)*879.6;
            
            const distanceKm = (totalShuttlesCompleted * 20) / 1000;
            document.getElementById('distance-txt').innerText = distanceKm.toFixed(2) + " km";
        }

        function finishTest() {
            music.pause(); music.currentTime = 0;
            clearInterval(sTimer); clearInterval(globalTimer);
            window.speechSynthesis.cancel();
            const [bestL, bestS] = (localStorage.getItem('beep_record_' + currentCat) || "1:1").split(':').map(Number);
            if (level > bestL || (level == bestL && shuttle > bestS)) {
                localStorage.setItem('beep_record_' + currentCat, `${level}:${shuttle}`);
                document.getElementById('new-record-banner').style.display = 'block';
            }
            document.getElementById('final-display').innerText = `${level}:${shuttle}`;
            document.getElementById('overlay').style.display = 'flex';
        }

        function resetAllRecords() {
            if (confirm("Reset all category PBs?")) {
                ['JNR_GIRLS', 'JNR_BOYS', 'SNR_GIRLS', 'SNR_BOYS', 'OPEN_AGES'].forEach(cat => localStorage.removeItem('beep_record_' + cat));
                initRecords();
            }
        }
    </script>
